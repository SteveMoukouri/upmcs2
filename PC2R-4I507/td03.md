```js
let compteur = ref 0;
let cle = Mutex.create();

let rec entree nb =
    if nb > 0 then
        begin
            Mutex.lock cle
            compteur := !compteur + 1
            Mutex.unlock cle
            entrer (nb - 1)
        end

let sortie  =
    while true do
        if !compteur > 0 then
            begin
                Mutex.lock cle
                compteur := compteur - 1
                Mutex.unlock cle
            end
```
